/** * ======================================================== * 👶 BB養成計劃 2.1 — 核心遊戲引擎 (Gameplay & Ending Engine) * ======================================================== */ // 1. 定義 BB 的初始狀態與數據結構 class Baby { constructor(name) { this.name = name; this.age = 0; // 當前歲數 (0-18) this.educationFund = 1000; // 初始教育基金 // 常規六維屬性 (上限預設 100) this.stats = { iq: 20, // 🧠 智力 cre: 20, // 🎨 創意 fin: 20, // 📈 財商 phy: 20, // 💪 體魄 eq: 20, // ❤️ 情商 app: 20 // ✨ 外在 }; // 隱藏屬性上限 (預設全 100) this.statLimits = { iq: 100, cre: 100, fin: 100, phy: 100, eq: 100, app: 100 }; // 隱藏屬性計分器 (Hidden Counter / 鑰匙碎片) this.hiddenCounters = { science: 0, // 科研天賦進度 combat: 0, // 極限體能進度 rebel: 0 // 反叛/黑化值 }; // 已覺醒的隱藏天賦標籤 this.unlockedTraits = []; } // 輔助方法：安全的屬性增加（包含動態邊際遞減與上限控制） addStat(statName, amount) { if (!this.stats.hasOwnProperty(statName)) return; let current = this.stats[statName]; let limit = this.statLimits[statName]; let actualGain = amount; // 實施動態邊際遞減 if (current > 70) { actualGain = Math.round(amount * 0.4); // 71-100 瓶頸期，效果大減 } else if (current > 40) { actualGain = Math.round(amount * 0.7); // 41-70 熟練期，微幅衰減 } // 確保不超過當前上限 this.stats[statName] = Math.min(limit, current + actualGain); console.log(`[屬性變動] ${statName}: +${actualGain} (當前: ${this.stats[statName]}/${limit})`); } } // 2. 日常關卡引擎（依據歲數階段與 S/A/B 評級進行多維加點） class DailyMinigameEngine { static play(baby, stage, choice) { // 隨機模擬手操評級 S, A, B, C const ratings = ['S', 'A', 'B', 'C']; const playerRating = ratings[Math.floor(Math.random() * ratings.length)]; console.log(`\n🎮 --- 進行日常關卡 ---`); console.log(`年齡階段: ${stage} | 選擇模式: ${choice} | 玩家手操評級: [${playerRating}]`); // 根據評級決定加點數量與擴散維度 if (playerRating === 'S') { // S級：加3個維度 if (choice === '文科') { baby.addStat('iq', 5); baby.addStat('cre', 4); baby.addStat('fin', 4); } else { // 武科 baby.addStat('phy', 5); baby.addStat('eq', 4); baby.addStat('app', 4); } baby.educationFund += 200; // S級獎金 } else if (playerRating === 'A') { // A級：加2個維度 if (choice === '文科') { baby.addStat('iq', 4); baby.addStat('cre', 3); } else { baby.addStat('phy', 4); baby.addStat('eq', 3); } baby.educationFund += 120; } else if (playerRating === 'B') { // B級：只加1個主維度 if (choice === '文科') { baby.addStat('iq', 3); } else { baby.addStat('phy', 3); } baby.educationFund += 60; } else { // C級：懲罰保底 if (choice === '文科') baby.addStat('iq', 1); else baby.addStat('phy', 1); baby.educationFund += 20; } } } // 3. 每年人生選擇題與隱藏屬性突破引擎 class LifeOptionEngine { static triggerYearlyOption(baby, age) { console.log(`\n🔮 --- 觸發 ${age} 歲人生選擇題 ---`); // 5歲：幼稚園面試 (文武分水嶺) if (age === 5) { console.log(`問題: 面試官問想對小鳥做什麼？`); // 模擬玩家選擇 A: 計下飛幾遠 console.log(`玩家選擇: [A. 計下佢每日飛幾遠]`); baby.addStat('iq', 5); baby.hiddenCounters.science += 2; // 後台隱藏種子 } // 11歲：呈分試打機風波 (隱藏屬性關鍵前置) else if (age === 11) { console.log(`問題: BB用教育基金買二手電腦幫同學寫Code/代打賺錢？`); // 模擬玩家選擇 B: 保留電腦並指導寫Code console.log(`玩家選擇: [B. 保留電腦，要求每日限時研究寫Code]`); baby.addStat('cre', 5); baby.addStat('fin', 5); baby.hiddenCounters.science += 3; } // 15歲：黑客風波 (性格黑化分支) else if (age === 15) { console.log(`問題: BB駭進學校改全班分數，一臉驕傲。你點做？`); // 模擬玩家選擇 B: 體諒並引導正軌 (若選痛打會加 rebel 導致黑化結局) console.log(`玩家選擇: [B. 拍拍肩膀，引導認識網絡安全]`); baby.addStat('eq', 5); baby.hiddenCounters.science += 2; } // 16歲：進行隱藏屬性突破檢查 (Cap Break Check) if (age === 16) { this.checkCapBreak(baby); } } // 檢查是否滿足「打破常規100限制，衝向150」的條件 static checkCapBreak(baby) { console.log(`\n🔒 [上限突破檢查] 正在檢測BB是否能打破凡人極限...`); // 智力上限突破條件：科研種子累積 >= 5 且 基礎智力 >= 70 (經遞減後的實際分數) if (baby.hiddenCounters.science >= 5 && baby.stats.iq >= 60) { baby.statLimits.iq = 150; // 上限粉碎！ baby.unlockedTraits.push("【絕對真理之眼】"); console.log(`🌟 💥 嘭！BB 覺醒了隱藏天賦: 【絕對真理之眼】！智力上限解鎖至 150 點！`); } else { console.log(`檢查完畢：BB 資質平庸或家長培育方向不夠專一，維持普通上限。`); } } } // 4. 終局判定引擎（多重條件權重陣列：三維 -> 雙維 -> 單維 -> 保底） class EndingDirector { static determineEnding(baby) { console.log(`\n========================================================`); console.log(`🏁 18年期滿！【${baby.name}】迎來人生終局結算面板`); console.log(`最終六維數據:`, baby.stats); console.log(`解鎖隱藏標籤:`, baby.unlockedTraits); console.log(`========================================================`); const s = baby.stats; const t = baby.unlockedTraits; // 提取滿足不同分數線的屬性陣列 let over70 = Object.keys(s).filter(k => s[k] >= 70); let over75 = Object.keys(s).filter(k => s[k] >= 75); let over85 = Object.keys(s).filter(k => s[k] >= 85); // ------------------ 第一層：神級/暗黑特殊結局 ------------------ if (s.iq >= 120 && t.includes("【絕對真理之眼】") && s.eq >= 60) { return { title: "🧬 諾貝爾物理學獎得主", desc: "人類群星閃耀時！BB 破解了暗物質物理公式，在斯德哥爾摩接過獎盃。致辭中他特別感謝你 11 歲時沒有砸爛他的二手電腦。" }; } // ------------------ 第二層：三維共鳴流派 (任三維 >= 70) ------------------ if (over70.length >= 3) { if (over70.includes('iq') && over70.includes('fin') && over70.includes('eq')) { return { title: "👑 頂級科技獨角獸 CEO", desc: "全能型跨界領袖。BB 不僅有研發 AI 的高智商，還懂得跟 VC 融資，更能完美管理千人團隊。公司於 18 歲正式掛牌上市！" }; } if (over70.includes('phy') && over70.includes('app') && over70.includes('cre')) { return { title: "🎭 荷里活動作巨星 / 傳奇舞者", desc: "強大的肉體、迷人的骨相與天馬行空的創意完美結合，全球影迷為之瘋狂。" }; } } // ------------------ 第三層：雙維共鳴流派 (任兩維 >= 75) ------------------ if (over75.length >= 2) { if (over75.includes('iq') && over75.includes('fin')) { return { title: "📊 華爾街頂級對沖基金經理", desc: "金錢永不眠。BB 利用極致的數學模型和對金錢的敏銳度做空全球，成為金融海嘯中的收割機器。" }; } if (over75.includes('phy') && over75.includes('eq')) { return { title: "🚒 英雄紀律部隊指揮官", desc: "擁有火海逆行的體能與支撐戰友的強大內心，成為受全民愛戴的救災英雄。" }; } } // ------------------ 第四層：單維極致流派 (任一維 >= 85) ------------------ if (over85.length >= 1) { if (over85.includes('iq')) return { title: "🧬 國家科學院院士", desc: "解開了困擾人類百年的數學難題，名字永遠留在教科書上。" }; if (over85.includes('phy')) return { title: "🥊 傳奇重量級拳王", desc: "肉體凡胎的巔峰，在賽場上用純粹的物理力量與速度讓對手絕望。" }; if (over85.includes('fin')) return { title: "🪙 連鎖商業房產大亨", desc: "18歲已靠投資和商道，賺到常人幾輩子花不完的資產。" }; } // ------------------ 第五層：養成失敗與水桶普通人保底 ------------------ let totalScore = Object.values(s).reduce((a, b) => a + b, 0); if (totalScore <= 180) { return { title: "🛋️ 終極啃老族", desc: "過低的屬性和擺爛的心態讓 BB 失去動力。18歲後拒絕踏出房門，每天在家打機等發零用錢。" }; } // 最終水桶人保底 return { title: "👔 朝九晚五的辦公室螺絲釘", desc: "平凡亦是福。成為一名安穩的會計或HR，每天下班買菜回家陪你食飯，過著溫馨平淡的一生。" }; } } /** * ======================================================== * 🚀 模擬 18 年遊戲完整運行 (Gameplay Simulation Loop) * ======================================================== */ const myBaby = new Baby("小明"); console.log(`👶 遊戲開始！你迎來了可愛的 BB 【${myBaby.name}】`); for (let year = 1; year <= 18; year++) { myBaby.age = year; // 判定年齡日常關卡類別 let stage = ""; let choice = "文科"; // 玩家偏向走文科智力流派 if (year <= 3) { stage = "🍼 嬰幼兒期日常"; choice = "武科"; // 幼兒期強制以生理體魄為主 } else if (year <= 6) { stage = "🎒 幼兒園期日常"; choice = "文科"; // 玩家主動選擇文科 } else if (year <= 12) { stage = "🏫 小學期日常"; choice = "文科"; // 繼續專攻文科 } else { stage = "⚡ 中學青春期日常"; choice = "文科"; // 衝刺文科 } // 1. 每年親自手操打日常關 DailyMinigameEngine.play(myBaby, stage, choice); // 2. 觸發特定歲數的人生選擇題 if ([5, 11, 15, 16].includes(year)) { LifeOptionEngine.triggerYearlyOption(myBaby, year); } } // 3. 18年期滿，呼叫終局判定引擎 const finalResult = EndingDirector.determineEnding(myBaby); console.log(`\n🎉 【最終結局解鎖】 🎉`); console.log(`👑 職業稱號: ${finalResult.title}`); console.log(`📝 人生軌跡: ${finalResult.desc}\n`);
